Einrichtung in 3 Schritten:

1. Erstelle ein neues Google Sheet in eurem Vereins-Drive mit den Spalten:
Timestamp, Angler, Datum, Uhrzeit, Gewässer, Fischart, Größe_cm, Gewicht_kg, Köder, Bemerkungen, Foto_URL

2. Öffne Erweiterungen → Apps Script im Sheet und füge das Skript (unten) ein. Deploye es als Web App (Zugriff: Jeder).

3. Kopiere die Script-URL hier rein und klicke "Speichern".
📜 Google Apps Script Code
const SHEET_NAME = 'Fangbuch'; function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let ws = ss.getSheetByName(SHEET_NAME); if (!ws) { ws = ss.insertSheet(SHEET_NAME); ws.appendRow(['Timestamp','Angler','Datum','Uhrzeit', 'Gewässer','Fischart','Größe_cm','Gewicht_kg', 'Köder','Bemerkungen','Foto_URL']); } const d = JSON.parse(e.postData.contents); ws.appendRow([new Date(), d.angler, d.datum, d.uhrzeit, d.gewaesser, d.fischart, d.groesse, d.gewicht, d.koeder, d.bemerkungen, d.foto_url || '']); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const ws = ss.getSheetByName(SHEET_NAME); if (!ws) return json({rows:[]}); const rows = ws.getDataRange().getValues(); const headers = rows.shift(); return ContentService .createTextOutput(JSON.stringify({ ok:true, headers, rows })) .setMimeType(ContentService.MimeType.JSON); }
